# 38 new Diff_ columns from T1/T2 pairs
for t1_col in t1_cols:
    stat = t1_col[3:]  # strip "T1_"
    t2_col = f'T2_{stat}'
    diff_col = f'Diff_{stat}'
    if t2_col in mm.columns:
        mm[diff_col] = mm[t1_col] - mm[t2_col]

Elo Win Probability
Computed with home-court advantage (HCA = 100 Elo points):
hca = where(Is_Home == 1, 100, -100)
Elo_Win_Prob = 1 / (1 + 10 ** (-(Diff_Elo + hca) / 400))
